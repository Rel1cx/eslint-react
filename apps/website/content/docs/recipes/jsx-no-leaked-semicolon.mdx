---
title: JSX No Leaked Semicolon
description: Detects trailing semicolons that would be rendered as text nodes in JSX.
---

## Overview

When refactoring JSX, trailing semicolons may be accidentally left immediately after JSX elements or fragments. For example, when moving a standalone expression like `return <div />;` inside another JSX element, the trailing semicolon can come along and end up as a child of the parent element. This causes `;` to be unexpectedly rendered as a visible text node in the output.

## Rule

Copy the following code into your project (e.g. `.config/jsxNoLeakedSemicolon.ts`):

```ts title=".config/jsxNoLeakedSemicolon.ts"
import type { RuleFunction } from "@eslint-react/kit";
import type { TSESTree } from "@typescript-eslint/utils";

/** Detect trailing semicolons that would be rendered as text nodes in JSX. */
export function jsxNoLeakedSemicolon(): RuleFunction {
  function isJSXElementLike({ type }: TSESTree.Node) {
    return type === "JSXElement" || type === "JSXFragment";
  }

  function hasLeakedSemicolon(text: string) {
    return text.startsWith(";\n") || text.startsWith(";\r");
  }

  return (context) => {
    function check(node: TSESTree.JSXText | TSESTree.Literal) {
      if (!isJSXElementLike(node.parent)) return;
      if (!hasLeakedSemicolon(context.sourceCode.getText(node))) return;
      context.report({
        node,
        message: "Potential leaked trailing semicolon. This semicolon will be rendered as a text node.",
      });
    }

    return {
      JSXText: check,
      Literal: check,
    };
  };
}
```

## Config

```ts title="eslint.config.ts"
import eslintReactKit from "@eslint-react/kit";
import { jsxNoLeakedSemicolon } from "./.config/jsxNoLeakedSemicolon";

export default [
  // ... other configs
  {
    ...eslintReactKit()
      .use(jsxNoLeakedSemicolon)
      .getConfig(),
    files: ["src/**/*.tsx"],
  },
];
```

## Examples

### Invalid

```tsx
function MyComponent() {
  return (
    <div>
      <div />;
    </div>
  );
}
```

```tsx
function MyComponent() {
  return (
    <div>
      <Component>
        <div />
      </Component>;
    </div>
  );
}
```

```tsx
function MyComponent() {
  return (
    <>
      <div />;
    </>
  );
}
```

### Valid

No trailing semicolon:

```tsx
function MyComponent() {
  return (
    <div>
      <div />
    </div>
  );
}
```

Semicolon on its own line — intentional text:

```tsx
function MyComponent() {
  return (
    <div>
      <div />
      ;
    </div>
  );
}
```

Semicolon as string expression — intentional:

```tsx
function MyComponent() {
  return (
    <div>
      <div />{';'}
    </div>
  );
}
```

Return statement ending with semicolon — valid JavaScript, not inside JSX:

```tsx
function MyComponent() {
  return <div />;
}
```
