Variable resolution and assignment tracking utilities for ESLint rules.

## Installation

```sh
npm install --save-dev @eslint-react/var
```

## Overview

The `@eslint-react/var` package provides utilities for:

- **Variable Resolution** - Resolve variable values and references
- **Assignment Tracking** - Track assignments to variables
- **Value Comparison** - Compare values and assignments
- **Object Type Computation** - Analyze object types and structures

## API Reference

### Variable Resolution

| Function                              | Description                                    |
| ------------------------------------- | ---------------------------------------------- |
| `resolve(node, context)`              | Resolve a variable to its definition and value |
| `resolveToValue(node, context)`       | Resolve a node to its actual value             |
| `findEnclosingAssignmentTarget(node)` | Find the enclosing assignment target           |

### Assignment Analysis

| Function                        | Description                                |
| ------------------------------- | ------------------------------------------ |
| `isAssignmentTargetEqual(a, b)` | Check if two assignment targets are equal  |
| `isValueEqual(a, b)`            | Check if two values are semantically equal |

### Object Type Computation

| Function                      | Description                           |
| ----------------------------- | ------------------------------------- |
| `computeObjectType(node)`     | Compute the type of an object literal |
| `getObjectPropertyType(node)` | Get the type of an object property    |

## Usage Examples

### Resolving a Variable

```ts
import { resolve } from "@eslint-react/var";
import type { TSESTree } from "@typescript-eslint/utils";

function getVariableValue(node: TSESTree.Identifier, context: RuleContext) {
  const resolution = resolve(node, context);

  if (resolution) {
    return {
      definition: resolution.definition,
      value: resolution.value,
      references: resolution.references,
    };
  }

  return null;
}
```

### Tracking Assignments

```ts
import { findEnclosingAssignmentTarget, isAssignmentTargetEqual } from "@eslint-react/var";

function analyzeAssignment(node: TSESTree.Node) {
  const target = findEnclosingAssignmentTarget(node);

  if (target) {
    // Check if this is assigning to a specific variable
    const isStateVariable = target.type === "Identifier" &&
      target.name.startsWith("set");

    return { target, isStateVariable };
  }

  return null;
}
```

### Comparing Values

```ts
import { isValueEqual } from "@eslint-react/var";

function checkDuplicateValues(nodes: TSESTree.Node[]) {
  const duplicates: Array<[TSESTree.Node, TSESTree.Node]> = [];

  for (let i = 0; i < nodes.length; i++) {
    for (let j = i + 1; j < nodes.length; j++) {
      if (isValueEqual(nodes[i], nodes[j])) {
        duplicates.push([nodes[i], nodes[j]]);
      }
    }
  }

  return duplicates;
}
```

### Computing Object Types

```ts
import { computeObjectType } from "@eslint-react/var";

function analyzeObjectLiteral(node: TSESTree.ObjectExpression) {
  const objectType = computeObjectType(node);

  return {
    hasSpread: objectType.hasSpread,
    properties: objectType.properties,
    isStable: objectType.isStable,
  };
}
```

## Integration with Other Utilities

The `var` module works well with the other utility modules:

```ts
import { isFunction } from "@eslint-react/ast";
import { resolve } from "@eslint-react/var";

function analyzeCallback(context, node) {
  // Resolve the variable
  const resolution = resolve(node, context);

  if (resolution?.value && isFunction(resolution.value)) {
    // The variable resolves to a function
    return analyzeFunction(resolution.value);
  }

  return null;
}
```
