  This plugin is for advanced use cases where you need to build custom ESLint rules tailored to your project's specific requirements.
</Callout>

## Installation

```bash copy
npm install --save-dev eslint-plugin-react-kit
```

## Quick Start

Define custom rules directly in your `eslint.config.ts`:

```ts copy filename="eslint.config.ts"
import js from "@eslint/js";
import { definePlugin, defineRuleListener } from "eslint-plugin-react-kit";
import { defineConfig } from "eslint/config";
import tseslint from "typescript-eslint";

export default defineConfig(
  {
    files: ["**/*.{ts,tsx}"],
    extends: [
      js.configs.recommended,
      tseslint.configs.recommended,
    ],
    plugins: {
      local: definePlugin(
        {
          name: "function-component-definition",
          make: (ctx, kit) => {
            const { api, visitor } = kit.getComponentCollector(ctx);

            return defineRuleListener(
              visitor,
              {
                "Program:exit"(program) {
                  for (const { node } of api.getAllComponents(program)) {
                    if (node.type === "ArrowFunctionExpression") continue;
                    ctx.report({
                      node,
                      message: "Function components must be defined with arrow functions.",
                    });
                  }
                },
              },
            );
          },
        },
      ),
    },
    rules: {
      "local/function-component-definition": "error",
    },
  },
]);
```

## Rule Structure

Each custom rule follows this structure:

```ts copy
import { defineRuleListener } from "eslint-plugin-react-kit";

{
  name: "rule-name",
  // Rule logic using collectors
  make: (ctx, kit) => {
    // Initialize collectors
    const { api, visitor } = kit.getComponentCollector(ctx);

    // Merge collector visitors with your custom visitors
    return defineRuleListener(
      visitor,
      {
        "Program:exit"(program) {
          // Access collected data via api
          for (const { node } of api.getAllComponents(program)) {
            // Report issues
            ctx.report({
              node,
              message: "Your error message",
            });
          }
        },
      },
    );
  },
}
```
