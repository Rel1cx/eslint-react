# no-complicated-conditional-rendering

## Rule category

Complexity.

## What it does

Prevents complicated conditional rendering in JSX.

## Examples

### Failing

```tsx twoslash
import React from "react";

interface ExampleProps {
  hideShapes: boolean;
  debugSvg: boolean;
}

function Example({ hideShapes, debugSvg }: ExampleProps) {
  return (
    <div>
      {hideShapes
        ? null
        : debugSvg
        ? "<ShapesWithSVGs />"
        : "<ShapesToDisplay />"}
      {/* Avoid complicated conditional rendering. Extract the logic into separate elements or components. */}
    </div>
  );
}
```

### Passing

```tsx twoslash
import React from "react";

interface ExampleProps {
  hideShapes: boolean;
  debugSvg: boolean;
}

function Example({ hideShapes, debugSvg }: ExampleProps) {
  // @annotate: Early return if nothing to render
  if (hideShapes) {
    return null;
  }

  return debugSvg ? "<ShapesWithSVGs />" : "<ShapesToDisplay />";
}
```
