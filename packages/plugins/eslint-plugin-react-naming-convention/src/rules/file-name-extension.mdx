**Features**

`⚙️`

## Description

Enforces consistent use of the JSX file extension.

## Rule Options

```ts
type Options = "always" | "as-needed" | { allow: "always" | "as-needed", extensions?: string[], ignoreFilesWithoutCode?: boolean };
```

### `allow`

When to allow a JSX filename extension.

Default: `"as-needed"`

### `extensions`

List of file extensions that should be treated as JSX files.

Default: `[".jsx", ".tsx"]`

### `ignoreFilesWithoutCode`

When set to `true`, this rule ignores files that do not contain JSX syntax.

Default: `true`

## Examples

### Failing

```tsx title="foo.tsx"
export const foo = 42;
```

### Passing

```tsx title="foo.ts"
export const foo = 42;
```

```tsx title="foo.tsx"
export const foo = <div>Hello, world!</div>;
```

## Configuration Examples

```js title="eslint.config.js"
// ...
export default [
  // ...
  {
    files: ["**/*.tsx"],
    rules: {
      "@eslint-react/naming-convention/file-name-extension": ["warn", "as-needed"]
    }
  }
];
```

```js title="eslint.config.js"
// ...
export default [
  // ...
  {
    files: ["**/*.tsx"],
    rules: {
      "@eslint-react/naming-convention/file-name-extension": ["warn", { "allow": "always" }]
    }
  }
];
```

```js title="eslint.config.js"
// ...
export default [
  // ...
  {
    files: ["**/*.tsx"],
    rules: {
      "@eslint-react/naming-convention/file-name-extension": ["warn", "always"],
    },
  },
  {
    files: ["src/**/use*.tsx"],
    rules: {
      "@eslint-react/naming-convention/file-name-extension": [
        "warn",
        "as-needed",
      ],
    },
  },
  {
    files: ["fixtures/**/*.tsx"],
    rules: {
      "@eslint-react/naming-convention/file-name-extension": "off",
    },
  },
];
```
