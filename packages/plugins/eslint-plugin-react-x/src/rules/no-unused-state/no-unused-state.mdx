## Description

Warns about unused class component state.

## Examples

### Failing

```tsx
import React from "react";

class MyComponent extends React.Component {
  // Unused
  state = {
    foo: 1,
  };

  render() {
    return null;
  }
}
```

### Passing

```tsx
import React from "react";

class MyComponent extends React.Component {
  state = {
    foo: 1,
  };

  render() {
    return this.state.foo;
  }
}
```
