## Rule Details

When a `useState` setter is called with an expression that directly references the corresponding state variable, it may cause stale state bugs. Using the callback form ensures the update is always based on the latest state value.

## Common Violations

### Invalid

```tsx
import { useState } from "react";

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <button onClick={() => setCount(count + 1)}>
      {/* ^^^^^^^^^^^^^^^^^^^^^^^^ Use the callback form of 'setCount' to ensure the update is based on the latest state. */}
      {count}
    </button>
  );
}
```

```tsx
import { useState } from "react";

function Toggle() {
  const [enabled, setEnabled] = useState(false);

  return (
    <button onClick={() => setEnabled(!enabled)}>
      {/* ^^^^^^^^^^^^^^^^^^^^^^^^^^ Use the callback form of 'setEnabled' to ensure the update is based on the latest state. */}
      {enabled ? "On" : "Off"}
    </button>
  );
}
```

### Valid

```tsx
import { useState } from "react";

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <button onClick={() => setCount((prev) => prev + 1)}>
      {count}
    </button>
  );
}
```

```tsx
import { useState } from "react";

function Component() {
  const [count, setCount] = useState(0);

  // Setting a constant value does not reference state
  return <button onClick={() => setCount(0)}>Reset</button>;
}
```

```tsx
import { useState } from "react";

function Component() {
  const [count, setCount] = useState(0);

  // Already using the callback form
  return (
    <button onClick={() => setCount((prev) => prev + 1)}>
      {count}
    </button>
  );
}
```
