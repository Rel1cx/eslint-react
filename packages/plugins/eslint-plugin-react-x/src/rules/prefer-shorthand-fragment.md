## Description

Enforces shorthand syntax for fragments.

A **safe** auto-fix is available for this rule.

## Examples

### Failing

```tsx
import React, { Fragment } from "react";

function MyComponent() {
  return (
    <Fragment>
      <button />
      <button />
    </Fragment>
  );
}
```

### Passing

```tsx
import React from "react";

function MyComponent() {
  return (
    <>
      <button />
      <button />
    </>
  );
}
```
